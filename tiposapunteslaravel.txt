<?php

namespace App\Http\Controllers;

use App\Models\TipoApuntes;
use Illuminate\Http\Request;

class TipoApuntesController extends Controller
{
    /**
     * Display a listing of the resource.
     *
     * @return \Illuminate\Http\Response
     */
    public function index()
    {
        $datos['tipoapuntes']=TipoApuntes::orderBy('id','ASC')->paginate(25);
        return view('tipoapuntes.index', $datos);
    }

    /**
     * Show the form for creating a new resource.
     *
     * @return \Illuminate\Http\Response
     */
    public function create()
    {
        return view('tipoapuntes.create');
    }

    /**
     * Store a newly created resource in storage.
     *
     * @param  \Illuminate\Http\Request  $request
     * @return \Illuminate\Http\Response
     */
    public function store(Request $request)
    {
        /*$datosTipoapuntes = request()->except('_token');
        TipoApuntes::insert($datosTipoapuntes);*/

        $request->validate([
            'nombre'=>'required',
        ]);

        $tipoapunte = new tipoapuntes([
            'nombre'=> $request->get('nombre'),
        ]);
        $tipoapunte->save();
        return redirect('tipoapuntes');

    }

    /**
     * Display the specified resource.
     *
     * @param  \App\Models\TipoApuntes  $tipoApuntes
     * @return \Illuminate\Http\Response
     */
    public function show(TipoApuntes $tipoApuntes)
    {
        //
    }

    /**
     * Show the form for editing the specified resource.
     *
     * @param  \App\Models\TipoApuntes  $tipoApuntes
     * @return \Illuminate\Http\Response
     */
    public function edit($id)
    {
        $tipoapuntes=TipoApuntes::findOrFail($id);
        return view('tipoapuntes.edit', compact('tipoapuntes'));
    }

    /**
     * Update the specified resource in storage.
     *
     * @param  \Illuminate\Http\Request  $request
     * @param  \App\Models\TipoApuntes  $tipoApuntes
     * @return \Illuminate\Http\Response
     */
    public function update(Request $request, $id)
    {
        $datosTipoapuntes = request()->except('_token','_method');
        TipoApuntes::where('id','=',$id)->update($datosTipoapuntes);

        $tipoapuntes=TipoApuntes::findOrFail($id);
        return view('tipoapuntes.edit', compact('tipoapuntes'));
    }

    /**
     * Remove the specified resource from storage.
     *
     * @param  \App\Models\TipoApuntes  $tipoApuntes
     * @return \Illuminate\Http\Response
     */
    public function destroy($id)
    {
        TipoApuntes::destroy($id);
        return redirect('tipoapuntes');
    }
}
